# Network config template
# Template used to configure new vm with static ip address

TYPE=Ethernet
BOOTPROTO=static
DEFROUTE=yes[IF GW IS ON OTHER DEVICE write no]
NAME=[YOUR NAME DEVICE]
DEVICE=[YOUR DEVICE]
ONBOOT=yes
IPADDR=[YOUR IP]
PREFIX=[YOUR SUBNET]
GATEWAY=[YOUR GW]
DNS1=[YOUR DNS_OPTIONAL]
